#include <stdio.h>
#include <Windows.h>
//Темп в Ульянновске летом
void maid() {
	int v; //Температура в ульяновске

	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	printf("Введите температуру: ");
	scanf_s("%d", &v);

	if (v <= 10) {
		printf("Мороз!");
	}
	if ((v > 10 ) && (v <= 15)) {
		printf("Холодно!");
	}
	if ((v > 15) && (v <= 25)) {
		printf("Тепло!");
	}
	if (v > 25) {
		printf("Жарко!");
	}
}
